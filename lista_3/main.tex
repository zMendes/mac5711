Algoritmo MaiorMenor (v, n)
    maior <- v[1]
    menor <- v[1]
    para i <- 2 ate n faca
        se v[i] > maior
            entao maior <- v[i]
            senao se v[i] < menor
                entao menor <- v[i]
    devolva maior, menor
        \end{lstlisting}
